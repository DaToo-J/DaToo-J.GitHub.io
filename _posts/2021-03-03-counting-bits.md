```python
class Solution:
    def countBits(self, num: int) -> List[int]:
        res = [0] * (num+1)
        for i in range(num+1):
            if i % 2 == 0:
                res[i] = res[i//2]
            else:
                res[i] = res[i-1] + 1
        return res
```
